**Goal:** SDD §13 P0 — 리포 구조 + pilot CLI/manifest + hooks + tracks.yaml + CLAUDE.md v1을 구현해, 게이트·승인 없는 단계 전이를 코드가 차단하는 "빈 파이프라인"을 우회 시도 테스트와 함께 완성한다.

**Tech Stack:** Python 3.11+ (표준 라이브러리 + PyYAML), pytest, argparse(CLI), Claude Code hooks(JSON stdin/stdout 프로토콜). Windows 우선(개발 PC), 경로 처리는 `pathlib` 일관.

원장의 계약: ① `{seq, ts, type, payload}` 1라인 1이벤트, seq는 1부터 단조 증가. ② **커밋 = 개행으로 종결된 라인**(WAL 레코드 종결자) — 이 정의를 읽기·쓰기가 공유한다. 읽기는 개행 없는 마지막 라인을 **완전한 JSON이라도** 커밋되지 않은 것으로 스킵하고(우연히 파싱 가능한 찢긴 꼬리에 의존하지 않는 결정론 기준), 커밋된 라인의 파싱 실패는 위치 무관 `CorruptLedgerError`(fail-closed) — append도 내부 read를 경유하므로 **손상된 원장 위에 이어 쓰는 것 자체가 거부**된다(복구 경로는 수동 복구뿐 — `pilot rebuild`도 같은 read를 경유하므로 손상 원장에선 동일하게 거부된다, 의도된 fail-closed). ③ append는 파일 끝이 개행이 아니면 마지막 개행까지 truncate 후 기록 — 읽기와 같은 커밋 정의라 "rebuild가 반영한 이벤트를 다음 append가 삭제"하는 되돌림이 구조적으로 불가능하다(Codex F1). seq는 마지막 **커밋** 라인 기준 +1. ④ 모든 원장·GateResult 쓰기는 **job별 OS 파일락으로 직렬화**된다 — 인간 채널(별도 터미널 `pilot approve`)과 에이전트 세션 훅(자동 G1)이 같은 job에 동시 기록해도 seq 단조·attempt 유일이 유지된다(Codex F2). OS 락은 프로세스 종료 시 자동 해제라 stale lock이 없다.

```python
import json
import pytest
from pathlib import Path
from pilot import store


@pytest.fixture
def job_dir(tmp_path):
    d = tmp_path / "jobs" / "J-TEST-001"
    d.mkdir(parents=True)
    return d


def test_append_assigns_monotonic_seq(job_dir):
    e1 = store.append_event(job_dir, "job_created", {"job_id": "J-TEST-001"})
    e2 = store.append_event(job_dir, "stage_start", {"stage": "S0"})
    assert (e1["seq"], e2["seq"]) == (1, 2)
    lines = (job_dir / "events.jsonl").read_text(encoding="utf-8").splitlines()
    assert [json.loads(l)["seq"] for l in lines] == [1, 2]
    assert all(set(json.loads(l)) == {"seq", "ts", "type", "payload"} for l in lines)


def test_read_events_roundtrip(job_dir):
    store.append_event(job_dir, "job_created", {"job_id": "J-TEST-001"})
    store.append_event(job_dir, "stage_start", {"stage": "S0"})
    events = store.read_events(job_dir)
    assert [e["type"] for e in events] == ["job_created", "stage_start"]


def test_torn_tail_is_tolerated_on_read_and_append(job_dir):
    store.append_event(job_dir, "job_created", {"job_id": "J-TEST-001"})
    ledger = job_dir / "events.jsonl"
    with ledger.open("a", encoding="utf-8") as f:
        f.write('{"seq": 2, "ts": "x", "type": "stage_')  # 크래시로 찢긴 꼬리
    events = store.read_events(job_dir)          # 꼬리 스킵
    assert [e["seq"] for e in events] == [1]
    e = store.append_event(job_dir, "stage_start", {"stage": "S0"})
    assert e["seq"] == 2                          # 커밋 라인 기준 +1
    assert [ev["seq"] for ev in store.read_events(job_dir)] == [1, 2]


def test_complete_json_without_newline_is_not_committed(job_dir):
    """커밋 = 개행 종결(계약 ②). 완전한 JSON이라도 개행 없는 꼬리는 read가 스킵하고
    append가 truncate한다 — 'rebuild 반영 후 append가 삭제'하는 되돌림 봉쇄(Codex F1)."""
    store.append_event(job_dir, "job_created", {"job_id": "J-TEST-001"})
    ledger = job_dir / "events.jsonl"
    with ledger.open("a", encoding="utf-8") as f:
        f.write('{"seq": 2, "ts": "x", "type": "stage_start", "payload": {}}')  # 개행 없음
    assert [e["seq"] for e in store.read_events(job_dir)] == [1]   # read도 미커밋 취급
    e = store.append_event(job_dir, "stage_start", {"stage": "S0"})
    assert e["seq"] == 2                          # 같은 정의 — seq 재사용이 아니라 최초 커밋
```

```python
"""Event ledger + manifest store. events.jsonl is the single source of truth;
manifest.json is a rebuildable cache (SDD §3.5)."""
from __future__ import annotations

import contextlib
import hashlib
import json
import os
from datetime import datetime, timezone
from pathlib import Path

LEDGER_NAME = "events.jsonl"
MANIFEST_NAME = "manifest.json"


class CorruptLedgerError(Exception):
    """Non-tail line failed to parse — ledger integrity is broken (fail-closed)."""


def _now() -> str:
    return datetime.now(timezone.utc).isoformat()


def _ledger_path(job_dir: Path) -> Path:
    return Path(job_dir) / LEDGER_NAME


def _parse_lines(raw: str) -> list[dict]:
    """Parse the ledger. Commit = newline-terminated line (contract ②):
    an unterminated tail is NEVER an event — even if it happens to parse —
    and a parse failure on a committed line is corruption (fail-closed).
    Split on the '\n' byte only (NOT str.splitlines, which also breaks on
    U+2028/U+2029/U+0085) so read matches the append-side byte definition —
    otherwise a committed valid line carrying those chars in free text would
    be misread as corruption (permanent stuck job, no auto recovery)."""
    parts = raw.split("\n")
    committed = len(parts) - 1  # split always yields one trailing element after the final '\n'
    events: list[dict] = []
    for i, line in enumerate(parts[:committed]):
        if not line.strip():
            continue
        try:
            events.append(json.loads(line))
        except json.JSONDecodeError:
            raise CorruptLedgerError(f"unparseable committed ledger line {i + 1}")
    return events


def read_events(job_dir: Path) -> list[dict]:
    path = _ledger_path(job_dir)
    if not path.exists():
        return []
    return _parse_lines(path.read_text(encoding="utf-8"))


@contextlib.contextmanager
def _job_lock(job_dir: Path):
    """Cross-process mutex per job (contract ④, Codex F2) — serializes the
    human approve channel vs. agent-session hooks. OS-level lock: auto-released
    on process death, so there is no stale-lock cleanup.
    한계(P0 수용): 대기에 총 타임아웃이 없다 — 임계구역이 짧아 실질 무해하나,
    총 대기 상한 + 오류 보고는 P3 운영 강화 항목."""
    lock_file = (Path(job_dir) / ".lock").open("a+b")
    try:
        if os.name == "nt":
            import msvcrt
            lock_file.seek(0)
            while True:
                try:
                    msvcrt.locking(lock_file.fileno(), msvcrt.LK_LOCK, 1)
                    break
                except OSError:
                    continue        # LK_LOCK gives up after ~10s — keep waiting
        else:
            import fcntl
            fcntl.flock(lock_file.fileno(), fcntl.LOCK_EX)
        yield
    finally:
        try:
            if os.name == "nt":
                import msvcrt
                lock_file.seek(0)
                msvcrt.locking(lock_file.fileno(), msvcrt.LK_UNLCK, 1)
            else:
                import fcntl
                fcntl.flock(lock_file.fileno(), fcntl.LOCK_UN)
        finally:
            lock_file.close()


def _append_event_unlocked(job_dir: Path, type: str, payload: dict) -> dict:
    """Inner append — caller MUST hold _job_lock (lock is not reentrant)."""
    path = _ledger_path(job_dir)
    if path.exists():
        raw = path.read_bytes()
        if raw and not raw.endswith(b"\n"):
            # uncommitted tail (contract ②/③) -> truncate to last committed line
            with path.open("r+b") as f:
                f.truncate(raw.rfind(b"\n") + 1)
    events = read_events(job_dir)
    seq = events[-1]["seq"] + 1 if events else 1
    event = {"seq": seq, "ts": _now(), "type": type, "payload": payload}
    with path.open("a", encoding="utf-8", newline="\n") as f:
        f.write(json.dumps(event, ensure_ascii=False) + "\n")
    return event


def append_event(job_dir: Path, type: str, payload: dict) -> dict:
    with _job_lock(job_dir):
        return _append_event_unlocked(job_dir, type, payload)
```

```python
def fold_events(events: list[dict], track_def: dict) -> dict:
    """Pure fold: ledger -> manifest. The ledger is truth; this is the cache builder."""
    manifest: dict = {}
    for ev in events:
        t, p, ts = ev["type"], ev["payload"], ev["ts"]
        if t == "job_created":
            manifest = {
                "job_id": p["job_id"], "track": p["track"],
                "problem_class": p["problem_class"], "mode": p["mode"],
                "requester": p["requester"], "request_text": p["request_text"],
                "current_stage": None, "stages": {},
                "milestones": {
                    s["milestone"]: {"status": "pending"}
                    for s in track_def["stages"] if "milestone" in s
                },
                "created_at": ts, "updated_at": ts,
            }
        elif t == "stage_start":
            stage = manifest["stages"].setdefault(p["stage"], {
                "status": "in_progress", "started_at": ts,
                "artifacts": [], "gates": {}, "self_corrections": 0,
            })
            stage["status"] = "in_progress"
            manifest["current_stage"] = p["stage"]
        elif t == "stage_complete":
            manifest["stages"][p["stage"]]["status"] = "completed"
            manifest["stages"][p["stage"]]["completed_at"] = ts
        if manifest:
            manifest["updated_at"] = ts
    return manifest


def write_manifest_atomic(job_dir: Path, manifest: dict) -> None:
    path = Path(job_dir) / MANIFEST_NAME
    tmp = path.with_suffix(".json.tmp")
    tmp.write_text(json.dumps(manifest, ensure_ascii=False, indent=2), encoding="utf-8")
    os.replace(tmp, path)  # atomic on same volume (Windows/POSIX)


def rebuild_manifest(job_dir: Path, track_def: dict) -> dict:
    manifest = fold_events(read_events(job_dir), track_def)
    write_manifest_atomic(job_dir, manifest)
    return manifest


def read_manifest(job_dir: Path) -> dict:
    return json.loads((Path(job_dir) / MANIFEST_NAME).read_text(encoding="utf-8"))
```

```python
def sha256_file(path: Path) -> str:
    h = hashlib.sha256()
    with Path(path).open("rb") as f:
        for chunk in iter(lambda: f.read(65536), b""):
            h.update(chunk)
    return "sha256:" + h.hexdigest()

               and e["payload"]["gate"] == gate and e["payload"]["target"] == target)
```

```python
"""Track definition loader — fail-closed schema validation (SDD §3.1)."""
from __future__ import annotations

from pathlib import Path

import yaml

KNOWN_GATES = {"G1", "G2", "G3"}
TRACKS_DIR = Path(__file__).parent / "tracks"
_FILE_BY_NAME = {"track1-structural": "track1.yaml"}


class TrackDefError(Exception):
    pass


def load_track(name: str) -> dict:
    filename = _FILE_BY_NAME.get(name)
    if filename is None:
        raise TrackDefError(f"unknown track: {name}")
    return load_track_file(TRACKS_DIR / filename)


def load_track_file(path: Path) -> dict:
    try:
        td = yaml.safe_load(Path(path).read_text(encoding="utf-8"))
    except Exception as exc:
        raise TrackDefError(f"cannot parse {path}: {exc}") from exc
    _validate(td)
    return td


def _validate(td: dict) -> None:
    for key in ("track", "stages", "tool_classes", "rules", "limits"):
        if key not in td:
            raise TrackDefError(f"missing top-level key: {key}")
    seen_ids: set[str] = set()
    for s in td["stages"]:
        for key in ("id", "name", "exit_gates"):
            if key not in s:
                raise TrackDefError(f"stage missing key {key}: {s}")
        if s["id"] in seen_ids:
            raise TrackDefError(f"duplicate stage id: {s['id']}")
        seen_ids.add(s["id"])
        bad = set(s["exit_gates"]) - KNOWN_GATES
        if bad:
            raise TrackDefError(f"unknown gates {bad} in stage {s['id']}")
    for r in td["rules"]:
        if r.get("tool_class") not in td["tool_classes"]:
            raise TrackDefError(f"rule references undefined tool_class: {r}")
    for key in ("self_correction_max", "same_error_signature_max"):
        if not isinstance(td["limits"].get(key), int):
            raise TrackDefError(f"limits.{key} must be int")
```

**fold 확장 규칙:**
- `milestone_requested {milestone}` → `milestones[Mx].status`가 `approved`가 아니면 `pending` 유지(멱등). rejected였다면 `pending`으로 복귀(§3.6 재요청).
- `approval {milestone, approver, reason}` → `status=approved, approver, reason, ts` 기록. **이미 approved면 무변경(멱등, §3.5)**.
- `rejection {milestone, approver, reason}` → `status=rejected` + **소속 단계 `in_progress` 복귀(§3.6 반려 전이)** + `current_stage`를 그 단계로 되돌림.
- `override {gate, target, reason, approver}` → manifest에 `overrides` 리스트 누적(전이 판정이 조회).
- `self_correction {stage, signature}` → `stages[stage].self_corrections += 1`.
- `stage_paused {stage, reason}` → `stages[stage].status = "paused"`.
- `human_intervention {kind, reason}` → fold는 무변경(지표 전용) — 단 이벤트 존재 자체가 원장에 남는다.

**전진 판정(§3.6 공식):**

> 전진 가능 = ① 현 단계의 모든 exit_gates에 대해 **관찰된 (게이트×target) 인스턴스 전수가 충족되고 최소 1개 존재** — 다른 target의 PASS가 실패한 실제 산출물의 기록을 대체할 수 없다(Codex F3) ∧ ② 각 인스턴스의 verdict ∉ {FAIL, ERROR} (override로 무효화된 FAIL 제외 — 단 override는 **기록 시점 sha가 현재 파일과 일치할 때만** 유효, 재수정 시 소멸·재승인 요구, Codex F4) ∧ (PASS는 target_sha가 대상 파일의 현재 해시와 일치할 때만 유효 — TOCTOU §3.4) ∧ ③ 단계에 마일스톤 지정 시 status=approved (milestone 모드에 한함 — self_service 면제)

```python
"""Stage transition semantics (SDD §3.6) — pure decision logic shared by
`pilot advance` and the PreToolUse hook. No file writes here."""
from __future__ import annotations

from dataclasses import dataclass, field
from pathlib import Path

from pilot import store

BLOCKING_VERDICTS = {"FAIL", "ERROR"}


@dataclass
class Decision:
    allowed: bool
    reasons: list[str] = field(default_factory=list)
    next_stage: str | None = None
    milestone_wait: str | None = None   # 게이트는 충족, 마일스톤만 대기


def _stage_def(track_def: dict, stage_id: str) -> dict:
    return next(s for s in track_def["stages"] if s["id"] == stage_id)


def _next_stage_id(track_def: dict, stage_id: str) -> str | None:
    ids = [s["id"] for s in track_def["stages"]]
    i = ids.index(stage_id)
    return ids[i + 1] if i + 1 < len(ids) else None


def _is_overridden(manifest: dict, gate: str, target: str, current_sha: str) -> bool:
    """override는 기록 시점 내용(sha)에 결합된다(Codex F4) — target이 재수정되면
    소멸해 재검증·재승인을 요구한다. '이 내용에 대한 이 판정이 오탐'이지
    '이 경로는 영구 면제'가 아니다."""
    return any(o["gate"] == gate and o["target"] == target
               and o.get("target_sha") == current_sha
               for o in manifest.get("overrides", []))


def gate_satisfied(manifest: dict, job_dir: Path, stage_id: str,
                   gate: str) -> tuple[bool, str]:
    """Stage-exit check for one gate: EVERY recorded (gate, target) instance of
    this stage must hold (verdict + current sha) and at least one must exist —
    a passing dummy target cannot displace a failing real artifact (Codex F3)."""
    instances = manifest["stages"].get(stage_id, {}).get("gates", {}).get(gate) or {}
    if not instances:
        return False, f"{stage_id}.{gate}: 게이트 미실행"
    for tgt, rec in instances.items():
        target = Path(job_dir) / tgt
        current_sha = store.sha256_file(target) if target.exists() else None
        if rec["verdict"] in BLOCKING_VERDICTS:
            if rec["verdict"] == "FAIL" and current_sha is not None \
                    and _is_overridden(manifest, gate, tgt, current_sha):
                continue
            return False, f"{stage_id}.{gate}[{tgt}]: verdict={rec['verdict']}"
        if current_sha is None:
            return False, f"{stage_id}.{gate}[{tgt}]: 대상 소실"
        if current_sha != rec["target_sha"]:
            return False, (f"{stage_id}.{gate}[{tgt}]: target_sha 불일치 — 게이트 기록 후 "
                           f"산출물이 수정됨, 재실행 필요(§3.4)")
    return True, ""


def can_advance(manifest: dict, track_def: dict, job_dir: Path) -> Decision:
    stage_id = manifest["current_stage"]
    if stage_id is None:
        return Decision(False, ["활성 단계 없음 — pilot new 직후 stage_start 필요"])
    sdef = _stage_def(track_def, stage_id)
    stage_state = manifest["stages"].get(stage_id, {})
    reasons: list[str] = []

    if stage_state.get("status") == "paused":
        reasons.append(f"{stage_id}: paused — 사람 호출 대기(자가복구 상한, §5)")

    for gate in sdef["exit_gates"]:                    # ①·②
        ok, why = gate_satisfied(manifest, job_dir, stage_id, gate)
        if not ok:
            reasons.append(why)

    milestone_wait = None
    milestone = sdef.get("milestone")                  # ③
    if milestone and manifest["mode"] == "milestone":
        status = manifest["milestones"].get(milestone, {}).get("status")
        if status != "approved":
            if not reasons:                             # 게이트는 전부 충족
                milestone_wait = milestone
            reasons.append(f"{milestone}: status={status} (approved 필요)")

    nxt = _next_stage_id(track_def, stage_id)  # None = 최종 단계(완료 처리, Task 8)

    return Decision(allowed=not reasons, reasons=reasons,
                    next_stage=nxt, milestone_wait=milestone_wait)
```

```python
"""pilot CLI — the ONLY writer of job state (SDD §3.2). Argparse, JSON-to-stdout."""
from __future__ import annotations

import argparse
import getpass
import json
import os
import sys
from datetime import datetime, timezone
from pathlib import Path

from pilot import store, track_defs, transition

# 에이전트 세션 환경 마커(2차 잠금, §6.1). Task 16 실증 후 목록 보정.
AGENT_SESSION_MARKERS = ("CLAUDECODE",)


def _err(msg: str) -> int:
    print(f"pilot: {msg}", file=sys.stderr)
    return 1


def _out(obj: dict) -> int:
    print(json.dumps(obj, ensure_ascii=False, indent=2))
    return 0


def _jobs_root(args) -> Path:
    if args.jobs_root:
        return Path(args.jobs_root)
    if os.environ.get("CAE_JOBS_ROOT"):
        return Path(os.environ["CAE_JOBS_ROOT"])
    return Path.cwd() / "jobs"


def _load_job(args) -> tuple[Path, dict, dict]:
    """job_dir, manifest(rebuilt fresh), track_def — 쓰기 명령 공통 진입점."""
    job_dir = _jobs_root(args) / args.job
    if not job_dir.exists():
        raise FileNotFoundError(f"job not found: {args.job}")
    events = store.read_events(job_dir)
    created = next((e for e in events if e["type"] == "job_created"), None)
    if created is None:
        raise ValueError(f"ledger has no job_created event: {args.job}")
    if created["payload"]["job_id"] != args.job:
        raise ValueError(f"job_id mismatch: dir={args.job} ledger={created['payload']['job_id']}")
    track_def = track_defs.load_track(created["payload"]["track"])
    manifest = store.rebuild_manifest(job_dir, track_def)
    return job_dir, manifest, track_def


def _autogen_job_id(jobs_root: Path) -> str:
    stamp = datetime.now(timezone.utc).strftime("%y%m%d")
    existing = {p.name for p in jobs_root.glob(f"J-{stamp}-*")}
    n = 1
    while f"J-{stamp}-{n:03d}" in existing:
        n += 1
    return f"J-{stamp}-{n:03d}"


def cmd_new(args) -> int:
    track_def = track_defs.load_track(args.track)          # TrackDefError → 상위에서 exit 1
    jobs_root = _jobs_root(args)
    jobs_root.mkdir(parents=True, exist_ok=True)
    job_id = args.job or _autogen_job_id(jobs_root)
    job_dir = jobs_root / job_id
    if job_dir.exists():
        return _err(f"job already exists: {job_id}")
    job_dir.mkdir()
    store.append_event(job_dir, "job_created", {
        "job_id": job_id, "track": args.track, "problem_class": args.problem_class,
        "mode": args.mode,
        "requester": {"name": args.requester_name, "role": args.requester_role},
        "request_text": args.request,
    })
    first_stage = track_def["stages"][0]["id"]
    store.append_event(job_dir, "stage_start", {"stage": first_stage})
    return _out(store.rebuild_manifest(job_dir, track_def))


def build_parser() -> argparse.ArgumentParser:
    p = argparse.ArgumentParser(prog="pilot")
    sub = p.add_subparsers(dest="cmd", required=True)

    def common(sp, job_required=True):
        sp.add_argument("--jobs-root", default=None)
        sp.add_argument("--job", required=job_required)

    sp = sub.add_parser("new")
    sp.add_argument("--track", required=True)
    sp.add_argument("--problem-class", required=True,
                    choices=["closed_form", "linear_static_complex",
                             "fatigue_load_history", "nonlinear_advanced"])
    sp.add_argument("--mode", required=True, choices=["milestone", "self_service"])
    sp.add_argument("--requester-name", required=True)
    sp.add_argument("--requester-role", required=True, choices=["cae", "design", "test"])
    sp.add_argument("--request", required=True)
    common(sp, job_required=False)
    sp.set_defaults(func=cmd_new)
    return p


def main(argv=None) -> int:
    parser = build_parser()
    args = parser.parse_args(argv)
    try:
        return args.func(args)
    except (track_defs.TrackDefError, store.CorruptLedgerError,
            FileNotFoundError, ValueError) as exc:
        return _err(str(exc))
```

```python
def run_verifier(gate: str, target: Path, job_dir: Path) -> dict:
    """Returns a GateResult body. Never raises: missing/crashing verifier -> ERROR."""
    import gates  # late import so tests can monkeypatch gates.VERIFIERS
    verifier = getattr(gates, "VERIFIERS", {}).get(gate)

                "checks": [{"id": "verifier-missing", "result": "error",

                            "actual": "none (P0 skeleton)", "source": "SDD §4.4/D6"}],
                "duration_s": 0.0}

        return result
    except Exception as exc:
        return {"verdict": "ERROR",
                "checks": [{"id": "verifier-crash", "result": "error",
                            "expected": "clean verifier run",
                            "actual": repr(exc), "source": "SDD §4.4/D6"}],
```

```python
def _milestone_request_pending(events: list[dict], milestone: str) -> bool:
    """최신 milestone_requested 이후 approval/rejection이 없으면 True(재기록 생략)."""
    state = None
    for e in events:
        p = e["payload"]
        if e["type"] == "milestone_requested" and p["milestone"] == milestone:
            state = "requested"
        elif e["type"] in ("approval", "rejection") and p.get("milestone") == milestone:
            state = "resolved"
    return state == "requested"


def cmd_advance(args) -> int:
    job_dir, manifest, track_def = _load_job(args)
    all_ids = [s["id"] for s in track_def["stages"]]
    if all(manifest["stages"].get(s, {}).get("status") == "completed" for s in all_ids):
        return _out({"advanced": False, "job_complete": True,
                     "reasons": ["job 이미 완료 — 전 단계 completed"]})
    decision = transition.can_advance(manifest, track_def, job_dir)
    stage = manifest["current_stage"]

    if decision.allowed:
        events = store.read_events(job_dir)
        rework = any(e["type"] == "stage_complete" and e["payload"]["stage"] == stage
                     for e in events)
        payload = {"stage": stage}
        if rework:
            payload["rework"] = True
        store.append_event(job_dir, "stage_complete", payload)
        out = {"advanced": True}
        if decision.next_stage:
            store.append_event(job_dir, "stage_start", {"stage": decision.next_stage})
            out["stage"] = decision.next_stage
        else:
            out["job_complete"] = True
        store.rebuild_manifest(job_dir, track_def)
        return _out(out)

    if decision.milestone_wait:
        events = store.read_events(job_dir)
        if not _milestone_request_pending(events, decision.milestone_wait):
            store.append_event(job_dir, "milestone_requested",
                               {"milestone": decision.milestone_wait})
            store.rebuild_manifest(job_dir, track_def)
    return _out({"advanced": False, "milestone_wait": decision.milestone_wait,
                 "reasons": decision.reasons})
```

```python
    return _out(store.rebuild_manifest(job_dir, track_def))


def cmd_intervene(args) -> int:
    if args.resume and _deny_agent_session():
        return _err("--resume(paused 해제)은 인간 전용이다 — 에이전트 세션에서 거부됨"
                    "(SDD 개정 노트 5, §6.1과 동일 잠금)")
    job_dir, manifest, track_def = _load_job(args)
    stage = manifest["current_stage"]
    if args.resume and manifest["stages"].get(stage, {}).get("status") != "paused":
        # 검증을 기록보다 앞에 — 실패한 no-op 명령이 개입 이벤트를 남기면
        # 무개입 완주율(§12)이 오염된다
        return _err(f"{stage}은 paused가 아니다 — --resume 불필요(이벤트 미기록)")
    store.append_event(job_dir, "human_intervention",
                       {"kind": "manual", "reason": args.reason})
    if args.resume:
        store.append_event(job_dir, "stage_start", {"stage": stage})
        # 재개 후 자가복구 상한 윈도는 이 stage_start부터 재계산된다(store.note_self_correction)
    return _out(store.rebuild_manifest(job_dir, track_def))


def cmd_rebuild(args) -> int:
    jobs_root = _jobs_root(args)
    job_dir = jobs_root / args.job
    if not job_dir.exists():
        return _err(f"job not found: {args.job}")
    events = store.read_events(job_dir)
    created = next((e for e in events if e["type"] == "job_created"), None)
    if created is None:
        # 복구 명령이 traceback으로 죽으면 안 된다 — 원장에 창세 이벤트가 없으면 명시 오류
        return _err(f"ledger has no job_created event: {args.job}")
    track_def = track_defs.load_track(created["payload"]["track"])
    return _out(store.rebuild_manifest(job_dir, track_def))
```

```python
import json
import subprocess
import sys
from pathlib import Path

REPO = Path(__file__).parent.parent
HOOKS = REPO / ".claude" / "hooks"


def run_hook(script: str, payload: dict, env_extra: dict | None = None):
    """훅 스크립트를 실제 프로토콜(stdin JSON)로 실행. (rc, parsed_stdout|None)"""
    import os
    env = {**os.environ, "CLAUDE_PROJECT_DIR": str(REPO)}
    for var in ("CAE_JOB_ID", "CAE_COPILOT_DEV_UNLOCK", "CAE_JOBS_ROOT"):
        env.pop(var, None)               # 개발자 셸의 우연한 설정이 테스트를 오염하지 않게
    env.update(env_extra or {})
    r = subprocess.run([sys.executable, str(HOOKS / script)],
                       input=json.dumps(payload), capture_output=True,
                       text=True, env=env)
    out = json.loads(r.stdout) if r.stdout.strip() else None
    return r.returncode, out


def decision(out) -> str | None:
    if out is None:
        return None
    return out["hookSpecificOutput"]["permissionDecision"]
```

```python
    if RESUME_RE.search(cmd):
        hc.deny("[SDD 개정 노트 5] paused 해제(intervene --resume)는 인간 전용이다 — "
                "자가복구 상한으로 멈춘 단계는 엔지니어 검토 후 별도 터미널에서 재개한다.")
    if G2_RECORD_RE.search(cmd):
        hc.deny("[SDD §4.2] G2 기록은 SubagentStop 훅 전용 경로다 — 에이전트의 직접 "
                "기록은 차단. gate2-reviewer 서브에이전트를 통해 검증을 수행하라.")
    if SOLVER_RE.search(cmd):
        _check_solver_requires()
    if OFFICIAL_RE.search(cmd):
        _check_official_report()
    hc.no_opinion()


def _load_state():
    """(manifest, track_def) — 어떤 예외든 호출부의 fail-closed로 흐른다."""
    from pilot import store, track_defs
    job_dir = hc.active_job_dir()
    if job_dir is None:
        hc.deny("[SDD §3.2] CAE_JOB_ID 미설정(또는 job 부재) — 게이트 대상 도구는 "
                "fail-closed로 차단된다. 세션 기동자가 CAE_JOB_ID를 설정해야 한다.")
    manifest = store.read_manifest(job_dir)
    track_def = track_defs.load_track(manifest["track"])
    return job_dir, manifest, track_def


def _rule_requires(track_def: dict, tool_class: str) -> dict:
    for r in track_def["rules"]:
        if r["tool_class"] == tool_class:
            return r.get("requires", {})
    return {}


def _check_solver_requires() -> None:
    from pilot import transition
    job_dir, manifest, track_def = _load_state()
    reasons = []
    if manifest["stages"].get(manifest["current_stage"], {}).get("status") == "paused":
        reasons.append(f"{manifest['current_stage']}: paused — 사람 호출 대기(§5)")
    req = _rule_requires(track_def, "solver_run")
    for inst in req.get("gates_passed", []):
        stage_id, gate = inst.split(".")
        ok, why = transition.gate_satisfied(manifest, job_dir, stage_id, gate)
        if not ok:
            reasons.append(why)
    if manifest["mode"] == "milestone":
        for m in req.get("milestones_approved", []):
            if manifest["milestones"].get(m, {}).get("status") != "approved":
                reasons.append(f"{m}: 미승인")
    if reasons:
        hc.deny("[SDD §5] 솔버 실행 차단 — 선행 조건 미충족:\n- "
                + "\n- ".join(reasons)
                + "\n게이트를 통과시키고(산출물 수정 → pilot gate-run), 마일스톤은 "
                  "엔지니어 승인을 요청하라.")


def _check_official_report() -> None:
    _, manifest, _ = _load_state()
    if manifest["mode"] == "self_service" or \
            manifest["milestones"].get("M3", {}).get("status") != "approved":
        hc.deny("[SDD §5/§6.2] --official 보고서는 M3 승인 필수 — 미승인·셀프서비스는 "
                "'사전검토용' 워터마크 빌드만 허용된다(--official 제거).")

HM_QUERY_FALLBACK = "mcp__hm__hm_get_*"   # job 미설정 시 조회 판정 폴백(§7.1 명명 규칙)


def check_hm(data: dict) -> None:
    tool = data["tool_name"]
    if hc.active_job_dir() is None:
        # §5 "조회는 상시 허용" — job 없이는 tracks yaml(tool_classes SSOT)을 읽을 수
        # 없으므로, 조회 명명 규칙 프리픽스 폴백으로 판정한다(정책: 계획 리뷰 확정).
        if fnmatch.fnmatch(tool, HM_QUERY_FALLBACK):
            hc.no_opinion()
        hc.deny("[SDD §3.2] CAE_JOB_ID 미설정 — 변형 도구는 fail-closed 차단"
                "(조회 hm_get_*만 허용). 세션 기동자가 CAE_JOB_ID를 설정해야 한다.")
    job_dir, manifest, track_def = _load_state()
    classes = track_def["tool_classes"]
    if any(fnmatch.fnmatch(tool, p) for p in classes.get("hm_query", [])):
        hc.no_opinion()                   # 조회는 상시 허용(§5)
    if any(fnmatch.fnmatch(tool, p) for p in classes.get("hm_mutate", [])):
        cur = manifest["current_stage"]
        if manifest["stages"].get(cur, {}).get("status") == "paused":
            hc.deny(f"[SDD §5] {cur}: paused — 사람 호출 대기")
        allowed = _rule_requires(track_def, "hm_mutate").get("stage_in", [])
        if cur not in allowed:
            hc.deny(f"[SDD §5] hm_mutate는 stage ∈ {allowed}에서만 허용 — 현재 {cur}. "
                    f"전처리 단계가 아니면 모델을 변형할 수 없다.")
        hc.no_opinion()
    hc.deny(f"[SDD §7.1] '{tool}'은 조회/변형 클래스에 미분류 — 전수 매핑 보증 위반, "
            f"fail-closed로 차단.")


def main() -> None:

    if tool in WRITE_TOOLS:

    elif tool == "Bash":

    elif tool.startswith("mcp__hm__"):

if __name__ == "__main__":
    # fail-closed 래퍼(§5·D6): stdin 파싱 실패까지 포함해 어떤 예외든 deny로 끝난다.
    # (비영 종료는 non-blocking이라 fail-open — 그 창을 없앤다)

        main()
```

```python
def note_self_correction(job_dir: Path, stage: str, signature: str | None,
                         limits: dict) -> str:
    """Record one self-correction; pause the stage when limits are hit (SDD §5).
    Returns "recorded" or "paused".
    상한 윈도 = 해당 단계의 최신 stage_start 이후(intervene --resume 재개 시 자연
    리셋 — SDD 개정 노트 5). 이미 paused면 상한 이벤트를 재발화하지 않는다(§12
    개입 이벤트 부풀림 방지 — fold는 멱등이지만 원장 오염 자체를 막는다)."""
    append_event(job_dir, "self_correction",
                 {"stage": stage, "signature": signature})
    window: list[dict] = []
    paused = False
    for e in read_events(job_dir):
        if e["payload"].get("stage") != stage:
            continue
        if e["type"] == "stage_start":
            window, paused = [], False
        elif e["type"] == "stage_paused":
            paused = True
        elif e["type"] == "self_correction":
            window.append(e)
    if paused:
        return "paused"
    consecutive = 0
    if signature:
        for e in reversed(window):
            if e["payload"].get("signature") == signature:
                consecutive += 1
            else:
                break
    if len(window) >= limits["self_correction_max"] or \
            consecutive >= limits["same_error_signature_max"]:
        append_event(job_dir, "stage_paused",
                     {"stage": stage, "reason": "self-correction limit (SDD §5)"})
        append_event(job_dir, "human_intervention",
                     {"kind": "auto_limit", "stage": stage,
                      "reason": f"{stage}: 자가복구 상한 도달 — 사람 호출"})
        return "paused"
    return "recorded"
```

```python
"""PostToolUse — auto G1 on deck/report writes + self-correction accounting (SDD §5)."""
from __future__ import annotations

import sys
from pathlib import Path

sys.path.insert(0, str(Path(__file__).parent))
import hook_common as hc

GATED_SUBDIRS = ("decks", "reports")


def main() -> None:
    data = hc.read_input()
    if data.get("tool_name") not in {"Edit", "Write", "MultiEdit"}:
        hc.no_opinion()
    fp = data["tool_input"].get("file_path")
    job_dir = hc.active_job_dir()
    if not fp or job_dir is None:
        hc.no_opinion()
    target = Path(fp).resolve()
    try:
        rel = target.relative_to(job_dir.resolve())
    except ValueError:
        hc.no_opinion()
    if not rel.parts or rel.parts[0] not in GATED_SUBDIRS:
        hc.no_opinion()

    from pilot import gate_registry, store, track_defs
    manifest = store.read_manifest(job_dir)
    track_def = track_defs.load_track(manifest["track"])
    stage = manifest["current_stage"]
    result = gate_registry.run_verifier("G1", target, job_dir)
    ev = store.record_gate_result(job_dir, "G1", target, stage, result)

    paused = ""
    if ev["payload"]["attempt"] >= 2:
        failed_ids = ",".join(c["id"] for c in result.get("checks", [])
                              if c.get("result") != "pass") or None
        state = store.note_self_correction(job_dir, stage, failed_ids,
                                           track_def["limits"])
        if state == "paused":
            paused = ("\n⛔ 자가복구 상한 도달 — 단계가 paused 됐다. 추가 수정을 멈추고 "
                      "엔지니어를 호출하라(SDD §5).")
    store.rebuild_manifest(job_dir, track_def)

    v = ev["payload"]["verdict"]
    if v in ("FAIL", "ERROR") or paused:
        checks = "\n".join(f"- {c['id']}: {c.get('actual', '')} (기준: {c.get('source', '')})"
                           for c in result.get("checks", []))
        hc.post_feedback(f"[G1 {v}] {rel.as_posix()} (attempt "
                         f"{ev['payload']['attempt']})\n{checks}{paused}")
    hc.no_opinion()


if __name__ == "__main__":
    try:
        main()
    except SystemExit:
        raise
    except Exception as exc:
        # PostToolUse는 차단 불가 — fail-closed 보고만(§5 래퍼)
        hc.post_feedback(f"[SDD D6] PostToolUse 훅 오류 — G1 자동 실행 실패: {exc!r}. "
                         f"pilot gate-run G1을 수동 실행해 기록을 남겨라.")
```

```python
                    done = all(s.get("status") == "completed"
                               for s in m.get("stages", {}).values())

                                     f"(단계 {m['current_stage']}, {m['track']})")
```

```python
def test_empty_pipeline_full_run(job, capsys, monkeypatch):
    tmp_path, job_dir = job
    monkeypatch.setattr(gates, "VERIFIERS", dict(PASS_ALL))

    def adv():
        return run_cli(capsys, ["advance", "--job", "J-1",
                                "--jobs-root", str(tmp_path)])[1]

    def g1(path_rel):
        p = job_dir / path_rel
        p.parent.mkdir(parents=True, exist_ok=True)
        p.write_text("content", encoding="utf-8")
        run_cli(capsys, ["gate-run", "G1", str(p), "--job", "J-1",
                         "--jobs-root", str(tmp_path)])
        return p

    def approve(m):
        assert cli.main(["approve", m, "--job", "J-1", "--jobs-root", str(tmp_path),
                         "--reason", "ok",
                         "--inbox-dir", str(tmp_path / "inbox")]) == 0

    (job_dir / "plan.md").write_text("p", encoding="utf-8")
    pass_g2(capsys, tmp_path)                            # S0.G2
    assert adv()["stage"] == "S1"
    g1("decks/model.inp")                                # S1.G1
    assert adv()["milestone_wait"] == "M1"
    approve("M1")
    assert adv()["stage"] == "S2"
    g1("decks/loaded.inp")                               # S2.G1
    pass_g2(capsys, tmp_path, "decks/loaded.inp")        # S2.G2
    assert adv()["milestone_wait"] == "M2"
    approve("M2")
    assert adv()["stage"] == "S3"
    assert adv()["stage"] == "S4"                        # S3: exit_gates 없음
    (job_dir / "results").mkdir(exist_ok=True)
    verdict_doc = g1("reports/judgement.md")             # S4 산출물
    pass_g2(capsys, tmp_path, "reports/judgement.md")    # S4.G2
    run_cli(capsys, ["gate-run", "G3", str(verdict_doc), "--job", "J-1",
                     "--jobs-root", str(tmp_path)])      # S4.G3 (더미 PASS)
    assert adv()["milestone_wait"] == "M3"
    approve("M3")
    assert adv()["stage"] == "S5"
    g1("reports/final.md")                               # S5.G1
    assert adv()["job_complete"] is True

    m = store.read_manifest(job_dir)
    assert all(s["status"] == "completed" for s in m["stages"].values())
    # 지표 원천 확인: 원장만으로 게이트 1회 통과율 산출 가능(§12)
    first_runs = [e for e in store.read_events(job_dir)
                  if e["type"] == "gate_run" and e["payload"]["attempt"] == 1]
    assert first_runs and all(e["payload"]["verdict"] == "PASS" for e in first_runs)
```
