# P1 구현 계획 — cae-copilot 전처리: hm_mcp + bridge vendor + G1 덱 린트 + S1·S2 Skill

> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** SDD §13 P1 — hm-assistant bridge를 vendor하고, 45개 도구를 노출하는 hm_mcp MCP 서버(thin 릴레이)와 G1 덱 린트 실검증기(.fem/.inp), S1·S2 Skill을 구현해 "전처리 구간이 실제로 게이팅되며 도는 파이프라인"을 완성한다. P0 이월 정비 4건(session_summary dedup, M1~M3)을 선행한다.

**Architecture:** Claude Code ←(MCP stdio)→ `cae_tools/hm_mcp`(신규, 로직 없는 릴레이) ←TCP JSON-RPC :9876(프로토콜 무변경)→ `cae_tools/hm_bridge`(hm-assistant vendor, HM 2026.1 내장 Python). 도구 노출명은 `hm_get_*`(조회)/`hm_create_*` 등(변형) 명명 규칙을 따르며 — 이 이름이 P0 훅 게이팅(track1.yaml `tool_classes` 패턴 `mcp__hm__hm_*`)의 지탱점이다 — 서버는 기동 시 전 도구의 클래스 매칭을 자체 검사하고 미분류가 있으면 기동을 거부한다(fail-closed). G1 덱 린트는 `gates.VERIFIERS["G1"]` 등록만으로 P0의 PostToolUse 자동 트리거·자가복구 계수·advance 판정에 즉시 연결된다. 테스트는 HyperMesh 없이 vendored mock 서버(canned RPC 응답)와 오류 주입 덱 코퍼스로 완결하고, 실기(HM 2026.1) 검증은 MANUAL_P1로 분리한다.

**Tech Stack:** Python 3.11+ (표준 라이브러리 + PyYAML + **`mcp` SDK 신규 추가**), pytest, Claude Code hooks/MCP(project-scope `.mcp.json`). Windows 우선, `pathlib` 일관.

```python
def note_self_correction(job_dir: Path, stage: str, signature: str | None,
                         limits: dict) -> str:
    """Record one self-correction; pause the stage when limits are hit (SDD §5).
    Returns "recorded" or "paused".
    상한 윈도 = 해당 단계의 최신 stage_start 이후(intervene --resume 재개 시 자연
    리셋 — §5 재개 의미론, P0 개정분). 이미 paused면 self_correction 자체를 기록하지 않는다
    (P1 M1 — post-pause 원장 증가 차단. F#1 하드컷이 훅 경유 수정을 이미 막으므로
    이 선체크는 store 수준의 방어적 이중화다)."""
    def _scan() -> tuple[list[dict], bool]:
        window: list[dict] = []
        paused = False
        for e in read_events(job_dir):
            if e["payload"].get("stage") != stage:
                continue
            if e["type"] == "stage_start":
                window, paused = [], False
            elif e["type"] == "stage_paused":
                paused = True
            elif e["type"] == "self_correction":
                window.append(e)
        return window, paused

    _, paused = _scan()
    if paused:
        return "paused"
    append_event(job_dir, "self_correction",
                 {"stage": stage, "signature": signature})
    window, _ = _scan()
    consecutive = 0
    if signature:
        for e in reversed(window):
            if e["payload"].get("signature") == signature:
                consecutive += 1
            else:
                break
    if len(window) >= limits["self_correction_max"] or \
            consecutive >= limits["same_error_signature_max"]:
        append_event(job_dir, "stage_paused",
                     {"stage": stage, "reason": "self-correction limit (SDD §5)"})
        append_event(job_dir, "human_intervention",
                     {"kind": "auto_limit", "stage": stage,
                      "reason": f"{stage}: 자가복구 상한 도달 — 사람 호출"})
        return "paused"
    return "recorded"
```

```python
"""hm_mcp 도구 레지스트리 — 명명 규칙이 훅 게이팅의 지탱점(SDD §7.1).

MCP 노출명(hm_*)은 track1.yaml tool_classes 패턴(`mcp__hm__hm_get_*` 등)과
정합해야 하고, RPC 메서드명은 bridge 프로토콜 무변경 원칙에 따라 원명 그대로다
(대응은 P1 계획의 매핑 테이블이 SSOT). 서버 기동 시 validate_registry()가
전 도구의 조회/변형 클래스 매칭을 검사하고, 미분류·불일치·중복이 있으면
기동을 거부한다(fail-closed — 미분류 도구는 훅 게이팅 밖 실행 경로가 된다).
"""
from __future__ import annotations

from dataclasses import dataclass
from typing import Callable

QUERY_PREFIXES = ("hm_get_",)
MUTATE_PREFIXES = ("hm_create_", "hm_set_", "hm_delete_", "hm_import_",
                   "hm_mesh_", "hm_export_")
EXPLICIT_MUTATE = frozenset({"execute_tcl_safe"})     # SDD §7.1 명시 등재


@dataclass(frozen=True)
class ToolSpec:
    name: str            # MCP 노출명 (Claude Code에는 mcp__hm__<name>으로 보인다)
    func: Callable       # plain function — 시그니처·docstring이 MCP 스키마가 된다
    klass: str           # "query" | "mutate" — 선언값(이름 파생값과 일치 강제)
    rpc: str             # bridge RPC 메서드(원명 — 프로토콜 무변경)


def classify(name: str) -> str | None:
    """이름 → 클래스. 어느 규칙에도 안 걸리면 None(=미분류)."""
    if name in EXPLICIT_MUTATE:
        return "mutate"
    if name.startswith(QUERY_PREFIXES):
        return "query"
    if name.startswith(MUTATE_PREFIXES):
        return "mutate"
    return None


def validate_registry(specs: list[ToolSpec]) -> None:
    """미분류·클래스 불일치·중복명 → RuntimeError(기동 거부, fail-closed)."""
    seen: set[str] = set()
    for s in specs:
        if s.name in seen:
            raise RuntimeError(f"hm_mcp: duplicate tool name {s.name!r}")
        seen.add(s.name)
        derived = classify(s.name)
        if derived is None:
            raise RuntimeError(
                f"hm_mcp: unclassified tool {s.name!r} — 조회/변형 프리픽스 규칙"
                f"(SDD §7.1)에 맞지 않는 이름은 기동을 거부한다(fail-closed)")
        if derived != s.klass:
            raise RuntimeError(
                f"hm_mcp: class mismatch for {s.name!r}: "
                f"declared={s.klass!r} derived={derived!r}")


def build_registry() -> list[ToolSpec]:
    """전 도구 스펙 수집 + 검증. 서버 기동·테스트의 공용 진입점."""
    from cae_tools.hm_mcp.tools import (geometry, loadbc, material, mesh,
                                        query, solver, tcl)
    specs = [*query.SPECS, *geometry.SPECS, *mesh.SPECS, *material.SPECS,
             *loadbc.SPECS, *solver.SPECS, *tcl.SPECS]
    validate_registry(specs)
    return specs
```

```python
"""RPC 호출 런타임 — 클라이언트 관리 + 에러 정규화(SDD §7.1).

클라이언트는 per-call 소켓 연결(원 프로토콜)이라 별도 재연결 로직이 없다 —
실패해도 다음 호출이 새로 연결한다. 접속 설정은 환경변수:
HM_BRIDGE_HOST(기본 localhost) / HM_BRIDGE_PORT(기본 9876) /
HM_BRIDGE_TIMEOUT_S(기본 60).

에러는 3패밀리로 정규화해 에이전트가 읽는 메시지 규격을 고정한다(개정 노트 7):
  [hm_mcp:connection] — bridge 미기동·연결 끊김 (+ 기동 안내)
  [hm_mcp:timeout]    — 응답 지연 (+ 타임아웃 상향 안내)
  [hm_mcp:rpc:<Type>] — bridge가 반환한 실행 오류(원 타입 보존)
"""
from __future__ import annotations

import os

from cae_tools.hm_mcp.client import HyperMeshRPCClient, HyperMeshRPCError

_client: HyperMeshRPCClient | None = None


def get_client() -> HyperMeshRPCClient:
    global _client
    if _client is None:
        _client = HyperMeshRPCClient(
            host=os.environ.get("HM_BRIDGE_HOST", "localhost"),
            port=int(os.environ.get("HM_BRIDGE_PORT", "9876")),
            timeout=float(os.environ.get("HM_BRIDGE_TIMEOUT_S", "60")),
        )
    return _client


def reset_client() -> None:
    """환경변수 변경·테스트 격리용 — 다음 호출이 새 설정으로 클라이언트 생성."""
    global _client
    _client = None


def call(method: str, params: dict | None = None):
    """RPC 중계 + 에러 정규화. 반환값은 bridge result 그대로(JSON 호환)."""
    try:
        return get_client().call(method, params or {})
    except HyperMeshRPCError as exc:
        raise RuntimeError(
            f"[hm_mcp:rpc:{exc.error_type}] {exc.message}") from exc
    except ConnectionRefusedError as exc:
        raise RuntimeError(
            f"[hm_mcp:connection] {exc} — HyperMesh 콘솔에서 bridge 기동 확인: "
            f"%run cae_tools/hm_bridge/startup.py") from exc
    except TimeoutError as exc:
        raise RuntimeError(
            f"[hm_mcp:timeout] {exc} — 장시간 작업은 HM_BRIDGE_TIMEOUT_S 상향"
        ) from exc
    except ConnectionError as exc:
        raise RuntimeError(f"[hm_mcp:connection] {exc}") from exc
```

"""hm_mcp — HyperMesh MCP 서버(thin 릴레이, SDD §7.1).

역할: 도구 스키마 노출·RPC 중계·에러 정규화·명명 규칙 기동검사. 로직 없음.

try:                                    # mcp SDK v2 (FastMCP → MCPServer 개명)
    from mcp.server import MCPServer

    """레지스트리 검증(fail-closed) 후 전 도구를 등록한 MCP 서버를 반환."""

        mcp.tool(name=spec.name)(spec.func)
    return mcp


def main() -> None:
    build_server().run()                # stdio (기본 transport)


if __name__ == "__main__":
    main()

                       yield_strength: float) -> dict:

    assigning a property/section.

def hm_create_nodeset_by_box(name: str, bbox: list[float]) -> dict:
    """Select nodes inside an axis-aligned box and register them as a named
    node set. Use when a BC is described by GEOMETRIC EXTENT (e.g., "fix
    the root face at x=0") — never ask for a face pick. Thin slabs are fine.

    Args:
        name: Node-set name (e.g., "ns_root", "ns_tip").
        bbox: [xmin, ymin, zmin, xmax, ymax, zmax] in mm.

    Args:

        r: Cylinder radius in mm. range: [axial_min, axial_max] in mm.

```python
def test_stdio_protocol_smoke(hm_mock):
    """python -m cae_tools.hm_mcp 를 실제 stdio(JSON-RPC, 개행 구분)로 기동해
    initialize → tools/list 왕복으로 45종 노출을 프로토콜 수준에서 검증한다.
    mcp SDK v1/v2 어느 쪽이 설치됐든 이 스모크가 최종 판정이다."""
    import json as _json
    import os
    import subprocess
    import sys

    proc = subprocess.Popen(
        [sys.executable, "-m", "cae_tools.hm_mcp"],
        stdin=subprocess.PIPE, stdout=subprocess.PIPE,
        stderr=subprocess.PIPE, text=True, encoding="utf-8",
        env={**os.environ})
    import threading
    watchdog = threading.Timer(30.0, proc.kill)   # 무응답 행(hang) 방지 —
    watchdog.start()                              # kill되면 readline이 EOF로 끊겨 assert
    try:
        def send(msg):
            proc.stdin.write(_json.dumps(msg) + "\n")
            proc.stdin.flush()

        def recv(want_id, limit=50):
            for _ in range(limit):
                line = proc.stdout.readline()
                if not line:
                    raise AssertionError(
                        f"server closed stdout; stderr={proc.stderr.read()[:2000]}")
                try:
                    m = _json.loads(line)
                except ValueError:
                    continue
                if m.get("id") == want_id:
                    return m
            raise AssertionError("no matching response")

        send({"jsonrpc": "2.0", "id": 1, "method": "initialize",
              "params": {"protocolVersion": "2025-06-18", "capabilities": {},
                         "clientInfo": {"name": "pytest", "version": "0"}}})
        assert "result" in recv(1)
        send({"jsonrpc": "2.0", "method": "notifications/initialized"})
        send({"jsonrpc": "2.0", "id": 2, "method": "tools/list", "params": {}})
        tools = recv(2)["result"]["tools"]
        names = {t["name"] for t in tools}
        assert len(names) == 45
        assert "hm_export_solver_deck" in names
        assert "execute_tcl_safe" in names
        assert "submit_analysis" not in names and "execute_tcl_raw" not in names
    finally:
        watchdog.cancel()
        proc.kill()
        proc.wait(timeout=10)
```

"""deck_lint.yaml 로더 — 기준의 이중 표현 중 기계용 (SDD §4.1).

경로 해석: CAE_HARNESS_DIR 환경변수(테스트 주입) → 리포 루트 harness/.

from __future__ import annotations

from pathlib import Path

import yaml

_REPO_ROOT = Path(__file__).resolve().parents[2]


def harness_dir() -> Path:

```python
"""덱 파서 공통 — 고정폭(.fem)·키워드(.inp) 라인 모델 (P1 Task 13).

파서는 판정하지 않는다 — 구조만 뽑고, 판정은 fem_lint/inp_lint 체크가 한다.
연속 카드('+'/'*' 시작 라인)는 dev 코퍼스 범위 밖 — 독립 카드로 통과시키며
체크들의 이름 매칭에서 자연히 무시된다(§F1 범위 주석 참조).
"""
from __future__ import annotations

import re
from dataclasses import dataclass, field

# Nastran 단축 지수: '7.85-9' → 7.85e-9, '2.1+11' → 2.1e11
_NASTRAN_EXP = re.compile(r"^([+-]?(?:\d+\.?\d*|\.\d+))([+-]\d+)$")


def nastran_float(token: str) -> float:
    """단축 지수 포함 float 파싱. 실패 시 ValueError —
    고정폭 필드 밀림 검출의 근거(§F1)."""
    tok = token.strip()
    m = _NASTRAN_EXP.match(tok)
    if m:
        return float(f"{m.group(1)}e{m.group(2)}")
    return float(tok)


def split_fixed(line: str, width: int = 8, count: int = 10) -> list[str]:
    """small-field 고정폭 분할 — 각 필드 strip, 부족한 꼬리는 빈 문자열."""
    padded = line.rstrip("\n").ljust(width * count)
    return [padded[i * width:(i + 1) * width].strip() for i in range(count)]


@dataclass(frozen=True)
class Card:
    name: str              # 대문자 카드명(fields[0])
    fields: list[str]
    line_no: int
    raw: str
    free: bool             # 자유형(콤마 구분) 여부


def fem_split_sections(text: str):
    """(케이스 컨트롤, 벌크) — 각각 (line_no, line) 목록.
    $ 주석·빈 줄 제거, BEGIN BULK에서 분할, ENDDATA에서 종료."""
    case: list[tuple[int, str]] = []
    bulk: list[tuple[int, str]] = []
    in_bulk = False
    for i, line in enumerate(text.splitlines(), start=1):
        s = line.strip()
        if not s or s.startswith("$"):
            continue
        u = s.upper()
        if u.startswith("BEGIN BULK"):
            in_bulk = True
            continue
        if u.startswith("ENDDATA"):
            break
        (bulk if in_bulk else case).append((i, line))
    return case, bulk


def fem_cards(bulk, width: int = 8) -> list[Card]:
    cards: list[Card] = []
    for no, line in bulk:
        if "," in line:                       # free-field
            fields = [f.strip() for f in line.split(",")]
            cards.append(Card(fields[0].upper(), fields, no, line, True))
        else:
            fields = split_fixed(line, width)
            cards.append(Card(fields[0].upper(), fields, no, line, False))
    return cards


@dataclass(frozen=True)
class Keyword:
    name: str                      # 대문자, '*' 포함 — 예: "*SOLID SECTION"
    params: dict
    data: list[str] = field(default_factory=list)
    line_no: int = 0


def inp_keywords(text: str) -> list[Keyword]:
    """.inp 키워드 블록 파싱 — '**' 주석 제거, KEY=VALUE 파라미터 대문자 키."""
    kws: list[Keyword] = []
    cur: Keyword | None = None
    for i, line in enumerate(text.splitlines(), start=1):
        s = line.strip()
        if not s or s.startswith("**"):
            continue
        if s.startswith("*"):
            parts = [p.strip() for p in s.split(",")]
            params: dict = {}
            for p in parts[1:]:
                if "=" in p:
                    k, v = p.split("=", 1)
                    params[k.strip().upper()] = v.strip()
                elif p:
                    params[p.upper()] = True
            cur = Keyword(parts[0].upper(), params, [], i)
            kws.append(cur)
        elif cur is not None:
            cur.data.append(s)
    return kws
```

```python
"""OptiStruct .fem 린트 — G1 덱 체크 6종(§4.1). 결정론 Python — LLM 없음.

체크: required-cards(§C1) / fixed-field-alignment(§F1) / unit-heuristic(§U1) /
link-integrity(§L1) / naming(§N1) / output-requests(§O1).
반환: GateResult body {"verdict","checks"} — duration_s는 gate_registry가 붙인다.
"""
from __future__ import annotations

import re
from pathlib import Path

from gates.g1_lint import parse
from gates.g1_lint.criteria import load_deck_criteria

# 고정폭 수치 필드(0-기점 인덱스, fields[0]=카드명): §F1 검사 대상
_NUMERIC_FIELDS = {
    "GRID": ((3, "X1"), (4, "X2"), (5, "X3")),
    "MAT1": ((2, "E"), (4, "NU"), (5, "RHO")),
}
_HMNAME = re.compile(r'^\$HMNAME\s+\w+\s+\d+\s*"([^"]*)"', re.M)


def _check(cid, ok, expected, actual, source):
    return {"id": cid, "result": "pass" if ok else "fail",
            "expected": expected, "actual": actual, "source": source}


def lint(target: Path) -> dict:
    crit = load_deck_criteria()["checks"]
    try:
        text = Path(target).read_text(encoding="utf-8", errors="replace")
    except OSError as exc:
        return {"verdict": "ERROR",
                "checks": [{"id": "io", "result": "error",
                            "expected": "readable deck file",
                            "actual": repr(exc), "source": "SDD §4.4/D6"}]}
    case, bulk = parse.fem_split_sections(text)
    cards = parse.fem_cards(bulk, width=crit["fem_fixed_field"]["width"])
    by_name: dict[str, list[parse.Card]] = {}
    for c in cards:
        by_name.setdefault(c.name, []).append(c)
    checks = []

    # §C1 필수 카드
    rc = crit["fem_required_cards"]
    missing = [k for k in rc["cards"] if k not in by_name]
    if not any(k in by_name for k in rc["element_cards"]):
        missing.append(f"요소({'/'.join(rc['element_cards'])})")
    checks.append(_check("required-cards", not missing,
                         f"{rc['cards']} + 요소 {rc['element_cards']}",
                         "모두 존재" if not missing else f"누락: {missing}",
                         rc["source"]))

    # §F1 고정폭 필드 정렬 — 수치 필드가 float으로 파싱되는가
    ff = crit["fem_fixed_field"]
    bad = []
    for name, slots in _NUMERIC_FIELDS.items():
        for c in by_name.get(name, []):
            if c.free:
                continue                        # 자유형 라인은 §F1 적용 밖
            for idx, label in slots:
                tok = c.fields[idx]
                if not tok:
                    continue                    # 공란 허용 필드(MAT1.G 등)
                try:
                    parse.nastran_float(tok)
                except ValueError:
                    bad.append(f"line {c.line_no} {name}.{label}={tok!r}")
    checks.append(_check("fixed-field-alignment", not bad,
                         "고정폭 수치 필드 전부 float 파싱",
                         "정렬 정상" if not bad else "; ".join(bad[:5]),
                         ff["source"]))

    # §U1 단위 휴리스틱 — MAT1 E·RHO
    u = crit["units"]
    unit_bad = []
    for c in by_name.get("MAT1", []):
        for idx, label, pred in (
                (2, "E", lambda v: u["E_min_mpa"] <= v <= u["E_max_mpa"]),
                (5, "RHO", lambda v: v <= u["rho_max_t_mm3"])):
            tok = c.fields[idx]
            if not tok:
                continue
            try:
                v = parse.nastran_float(tok)
            except ValueError:
                continue                        # §F1이 잡는다 — 이중 계상 방지
            if not pred(v):
                unit_bad.append(
                    f"line {c.line_no} MAT1.{label}={v!r} (SI 의심)")
    checks.append(_check("unit-heuristic", not unit_bad,
                         f"E∈[{u['E_min_mpa']},{u['E_max_mpa']}] MPa, "
                         f"ρ≤{u['rho_max_t_mm3']} t/mm³",
                         "mm-t-s 정합" if not unit_bad
                         else "; ".join(unit_bad[:5]),
                         u["source"]))

    # §L1 연결 무결성 — CTETRA.PID→PSOLID, PSOLID.MID→MAT1
    lk = crit["fem_links"]
    mat_ids = {c.fields[1] for c in by_name.get("MAT1", []) if c.fields[1]}
    ps_ids = {c.fields[1] for c in by_name.get("PSOLID", []) if c.fields[1]}
    link_bad = []
    for c in by_name.get("PSOLID", []):
        if c.fields[2] and c.fields[2] not in mat_ids:
            link_bad.append(
                f"line {c.line_no} PSOLID.MID={c.fields[2]} → MAT1 없음")
    for c in by_name.get("CTETRA", []):
        if c.fields[2] and c.fields[2] not in ps_ids:
            link_bad.append(
                f"line {c.line_no} CTETRA.PID={c.fields[2]} → PSOLID 없음")
    checks.append(_check("link-integrity", not link_bad,
                         "CTETRA.PID→PSOLID, PSOLID.MID→MAT1",
                         "연결 정상" if not link_bad
                         else "; ".join(link_bad[:5]),
                         lk["source"]))

    # §N1 명명 규칙 — $HMNAME 주석 이름(원문 스캔 — 주석은 섹션 분할이 걸러냄)
    nm = crit["naming"]
    pat = re.compile(nm["pattern"])
    names = _HMNAME.findall(text)
    name_bad = [n for n in names
                if not pat.match(n) or len(n) > nm["max_len"]]
    checks.append(_check("naming", not name_bad,
                         f"{nm['pattern']} (≤{nm['max_len']}자)",
                         (f"{len(names)}개 정합" if names and not name_bad
                          else "적용 대상 없음($HMNAME 부재)" if not names
                          else f"위반: {name_bad[:5]}"),
                         nm["source"]))

    # §O1 출력 요청 — 케이스 컨트롤
    orq = crit["fem_output_requests"]
    case_upper = [l.upper().lstrip() for _, l in case]
    miss_out = [r for r in orq["requests"]
                if not any(l.startswith(r) for l in case_upper)]
    checks.append(_check("output-requests", not miss_out,
                         f"케이스 컨트롤에 {orq['requests']}",
                         "요청 존재" if not miss_out else f"누락: {miss_out}",
                         orq["source"]))

    verdict = "PASS" if all(c["result"] == "pass" for c in checks) else "FAIL"
    return {"verdict": verdict, "checks": checks}
```

```python
"""Abaqus .inp 린트 — G1 덱 체크 5종(§4.1). 결정론 Python — LLM 없음.

체크: required-keywords(§C2, *STEP 쌍 포함) / link-integrity(§L1) /
unit-heuristic(§U1) / naming(§N1) / output-requests(§O1).
"""
from __future__ import annotations

import re
from pathlib import Path

from gates.g1_lint import parse
from gates.g1_lint.criteria import load_deck_criteria


def _check(cid, ok, expected, actual, source):
    return {"id": cid, "result": "pass" if ok else "fail",
            "expected": expected, "actual": actual, "source": source}


def _first_float(data_lines: list[str]):
    if not data_lines:
        return None
    tok = data_lines[0].split(",")[0].strip()
    try:
        return float(tok)
    except ValueError:
        return None


def lint(target: Path) -> dict:
    crit = load_deck_criteria()["checks"]
    try:
        text = Path(target).read_text(encoding="utf-8", errors="replace")
    except OSError as exc:
        return {"verdict": "ERROR",
                "checks": [{"id": "io", "result": "error",
                            "expected": "readable deck file",
                            "actual": repr(exc), "source": "SDD §4.4/D6"}]}
    kws = parse.inp_keywords(text)
    names = [k.name for k in kws]
    checks = []

    # §C2 필수 키워드 + *STEP/*END STEP 쌍
    rk = crit["inp_required_keywords"]
    missing = [k for k in rk["keywords"] if k not in names]
    if not ("*STEP" in names and "*END STEP" in names):
        missing.append("*STEP 쌍(*END STEP)")
    checks.append(_check("required-keywords", not missing,
                         f"{rk['keywords']} + *END STEP",
                         "모두 존재" if not missing else f"누락: {missing}",
                         rk["source"]))

    # §L1 섹션 연결 — *SOLID SECTION의 MATERIAL=·ELSET=
    lk = crit["inp_links"]
    mats = {k.params.get("NAME") for k in kws if k.name == "*MATERIAL"}
    elsets = {k.params.get("ELSET") for k in kws
              if k.name in ("*ELEMENT", "*ELSET") and k.params.get("ELSET")}
    link_bad = []
    sections = [k for k in kws if k.name == "*SOLID SECTION"]
    if not sections:
        link_bad.append("*SOLID SECTION 없음")
    for k in sections:
        if k.params.get("MATERIAL") not in mats:
            link_bad.append(f"line {k.line_no} MATERIAL="
                            f"{k.params.get('MATERIAL')} → *MATERIAL 없음")
        if k.params.get("ELSET") not in elsets:
            link_bad.append(f"line {k.line_no} ELSET="
                            f"{k.params.get('ELSET')} → ELSET 정의 없음")
    checks.append(_check("link-integrity", not link_bad,
                         "*SOLID SECTION의 MATERIAL→*MATERIAL, ELSET→정의",
                         "연결 정상" if not link_bad
                         else "; ".join(link_bad[:5]),
                         lk["source"]))

    # §U1 단위 휴리스틱 — *ELASTIC E·*DENSITY ρ (첫 데이터 값)
    u = crit["units"]
    unit_bad = []
    for k in kws:
        if k.name == "*ELASTIC":
            v = _first_float(k.data)
            if v is not None and not (u["E_min_mpa"] <= v <= u["E_max_mpa"]):
                unit_bad.append(f"line {k.line_no} *ELASTIC E={v!r} (SI 의심)")
        elif k.name == "*DENSITY":
            v = _first_float(k.data)
            if v is not None and v > u["rho_max_t_mm3"]:
                unit_bad.append(f"line {k.line_no} *DENSITY={v!r} (SI 의심)")
    checks.append(_check("unit-heuristic", not unit_bad,
                         f"E∈[{u['E_min_mpa']},{u['E_max_mpa']}] MPa, "
                         f"ρ≤{u['rho_max_t_mm3']} t/mm³",
                         "mm-t-s 정합" if not unit_bad
                         else "; ".join(unit_bad[:5]),
                         u["source"]))

    # §N1 명명 — NAME=·ELSET= 값
    nm = crit["naming"]
    pat = re.compile(nm["pattern"])
    seen = [v for k in kws for key, v in k.params.items()
            if key in ("NAME", "ELSET") and isinstance(v, str)]
    name_bad = [n for n in seen
                if not pat.match(n) or len(n) > nm["max_len"]]
    checks.append(_check("naming", not name_bad,
                         f"{nm['pattern']} (≤{nm['max_len']}자)",
                         f"{len(seen)}개 정합" if not name_bad
                         else f"위반: {name_bad[:5]}",
                         nm["source"]))

    # §O1 출력 요청 — 목록 중 1개 이상
    orq = crit["inp_output_keywords"]
    has_out = any(k.name in orq["keywords"] for k in kws)
    checks.append(_check("output-requests", has_out,
                         f"{orq['keywords']} 중 1개 이상",
                         "출력 요청 존재" if has_out else "출력 요청 없음",
                         orq["source"]))

    verdict = "PASS" if all(c["result"] == "pass" for c in checks) else "FAIL"
    return {"verdict": verdict, "checks": checks}
```

## 공통 규율 (5대 원칙 — hm_mcp 인자 검증·G1과 3중 집행)

2. **Node-based BC.** face-pick 없음 — 노드셋은 좌표로 만든다:
   `hm_get_bounding_box` → 좌표 계산 → `hm_create_nodeset_by_box`(또는 `_by_cylinder`).
3. **Volume-only tet.** `hm_mesh_tetmesh_component`의 volume_only 기본값(True)을 유지한다.
4. **RP(기준점)는 동적 계산.** 하드코딩 좌표 금지 — bbox에서 유도한다.
5. **Raw TCL 금지.** Python 도구 우선, 막히면 `execute_tcl_safe`(화이트리스트)가 최후 수단.
